#include<bits/stdc++.h>
int main()
{
    int arr[3],brr[3],a=0,b=0;
    for(int i=0;i<3;i++)
    {
        scanf("%d",&arr[i]);}
    for(int i=0;i<3;i++)
    {
        scanf("%d",&brr[i]);
    }
    for(int i=0;i<3;i++)
    {
        if(arr[i]>brr[i])
            a++;
        else if(arr[i]<brr[i])
            b++;
        else if(arr[i]==brr[i])
        {a=a;
        b=b;}
        
            
        
    }
    printf("%d %d",a,b);
    
    
    
}
